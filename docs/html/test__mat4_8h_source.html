329 TEST_IMPL(GLM_PREFIX, mat4_det) {
330  mat4 mat = GLM_MAT4_IDENTITY_INIT;
331  float t[6];
332  float a, b, c, d,
333  e, f, g, h,
334  i, j, k, l,
335  m, n, o, p;
336  float det1, det2;
337 
338  test_rand_mat4(mat);
339 
340  a = mat[0][0]; b = mat[0][1]; c = mat[0][2]; d = mat[0][3];
341  e = mat[1][0]; f = mat[1][1]; g = mat[1][2]; h = mat[1][3];
342  i = mat[2][0]; j = mat[2][1]; k = mat[2][2]; l = mat[2][3];
343  m = mat[3][0]; n = mat[3][1]; o = mat[3][2]; p = mat[3][3];
344 
345  t[0] = k * p - o * l;
346  t[1] = j * p - n * l;
347  t[2] = j * o - n * k;
348  t[3] = i * p - m * l;
349  t[4] = i * o - m * k;
350  t[5] = i * n - m * j;
351 
352  det1 = a * (f * t[0] - g * t[1] + h * t[2])
353  - b * (e * t[0] - g * t[3] + h * t[4])
354  + c * (e * t[1] - f * t[3] + h * t[5])
355  - d * (e * t[2] - f * t[4] + g * t[5]);
356 
357  det2 = GLM(mat4_det(mat));
358 
359  ASSERT(test_eq(det1, det2))
360 
361  TEST_SUCCESS
362 }
