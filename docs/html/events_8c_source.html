257 static size_t encode_utf8(char* s, unsigned int ch)
258 {
259  size_t count = 0;
260 
261  if (ch < 0x80)
262  s[count++] = (char) ch;
263  else if (ch < 0x800)
264  {
265  s[count++] = (ch >> 6) | 0xc0;
266  s[count++] = (ch & 0x3f) | 0x80;
267  }
268  else if (ch < 0x10000)
269  {
270  s[count++] = (ch >> 12) | 0xe0;
271  s[count++] = ((ch >> 6) & 0x3f) | 0x80;
272  s[count++] = (ch & 0x3f) | 0x80;
273  }
274  else if (ch < 0x110000)
275  {
276  s[count++] = (ch >> 18) | 0xf0;
277  s[count++] = ((ch >> 12) & 0x3f) | 0x80;
278  s[count++] = ((ch >> 6) & 0x3f) | 0x80;
279  s[count++] = (ch & 0x3f) | 0x80;
280  }
281 
282  return count;
283 }

400 static void key_callback(GLFWwindow* window, int key, int scancode, int action, int mods)
401 {
402  Slot* slot = glfwGetWindowUserPointer(window);
403  const char* name = glfwGetKeyName(key, scancode);
404 
405  if (name)
406  {
407  printf("%08x to %i at %0.3f: Key 0x%04x Scancode 0x%04x (%s) (%s) (with%s) was %s\n",
408  counter++, slot->number, glfwGetTime(), key, scancode,
409  get_key_name(key),
410  name,
411  get_mods_name(mods),
412  get_action_name(action));
413  }
414  else
415  {
416  printf("%08x to %i at %0.3f: Key 0x%04x Scancode 0x%04x (%s) (with%s) was %s\n",
417  counter++, slot->number, glfwGetTime(), key, scancode,
418  get_key_name(key),
419  get_mods_name(mods),
420  get_action_name(action));
421  }
422 
423  if (action != GLFW_PRESS)
424  return;
425 
426  switch (key)
427  {
428  case GLFW_KEY_C:
429  {
430  slot->closeable = !slot->closeable;
431 
432  printf("(( closing %s ))\n", slot->closeable ? "enabled" : "disabled");
433  break;
434  }
435 
436  case GLFW_KEY_L:
437  {
438  const int state = glfwGetInputMode(window, GLFW_LOCK_KEY_MODS);
439  glfwSetInputMode(window, GLFW_LOCK_KEY_MODS, !state);
440 
441  printf("(( lock key mods %s ))\n", !state ? "enabled" : "disabled");
442  break;
443  }
444  }
445 }

526  }
527  }

530  printf("%08x at %0.3f: Joystick %i was disconnected\n",
531  counter++, glfwGetTime(), jid);

535 int main(int argc, char** argv)
536 {
537  Slot* slots;
538  GLFWmonitor* monitor = NULL;
539  int ch, i, width, height, count = 1;
540 
541  glfwSetErrorCallback(error_callback);
542 
543  if (!glfwInit())
544  exit(EXIT_FAILURE);
545 
546  printf("Library initialized\n");
547 
548  glfwSetMonitorCallback(monitor_callback);
549  glfwSetJoystickCallback(joystick_callback);
550 
551  while ((ch = getopt(argc, argv, "hfn:")) != -1)
552  {
553  switch (ch)
554  {
555  case 'h':
556  usage();
557  exit(EXIT_SUCCESS);
558 
559  case 'f':
560  monitor = glfwGetPrimaryMonitor();
561  break;
562 
563  case 'n':
564  count = (int) strtoul(optarg, NULL, 10);
565  break;
566 
567  default:
568  usage();
569  exit(EXIT_FAILURE);
570  }
571  }
572 
573  if (monitor)
574  {
575  const GLFWvidmode* mode = glfwGetVideoMode(monitor);
576 
577  glfwWindowHint(GLFW_REFRESH_RATE, mode->refreshRate);
578  glfwWindowHint(GLFW_RED_BITS, mode->redBits);
579  glfwWindowHint(GLFW_GREEN_BITS, mode->greenBits);
580  glfwWindowHint(GLFW_BLUE_BITS, mode->blueBits);
581 
582  width = mode->width;
583  height = mode->height;
584  }
585  else
586  {
587  width = 640;
588  height = 480;
589  }
590 
591  slots = calloc(count, sizeof(Slot));
592 
593  for (i = 0; i < count; i++)
594  {
595  char title[128];
596 
597  slots[i].closeable = GLFW_TRUE;
598  slots[i].number = i + 1;
599 
600  snprintf(title, sizeof(title), "Event Linter (Window %i)", slots[i].number);
601 
602  if (monitor)
603  {
604  printf("Creating full screen window %i (%ix%i on %s)\n",
605  slots[i].number,
606  width, height,
607  glfwGetMonitorName(monitor));
608  }
609  else
610  {
611  printf("Creating windowed mode window %i (%ix%i)\n",
612  slots[i].number,
613  width, height);
614  }
615 
616  slots[i].window = glfwCreateWindow(width, height, title, monitor, NULL);
617  if (!slots[i].window)
618  {
619  free(slots);
620  glfwTerminate();
621  exit(EXIT_FAILURE);
622  }
623 
624  glfwSetWindowUserPointer(slots[i].window, slots + i);
625 
626  glfwSetWindowPosCallback(slots[i].window, window_pos_callback);
627  glfwSetWindowSizeCallback(slots[i].window, window_size_callback);
628  glfwSetFramebufferSizeCallback(slots[i].window, framebuffer_size_callback);
629  glfwSetWindowContentScaleCallback(slots[i].window, window_content_scale_callback);
630  glfwSetWindowCloseCallback(slots[i].window, window_close_callback);
631  glfwSetWindowRefreshCallback(slots[i].window, window_refresh_callback);
632  glfwSetWindowFocusCallback(slots[i].window, window_focus_callback);
