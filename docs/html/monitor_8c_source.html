40 // Lexically compare video modes, used by qsort
41 //
42 static int compareVideoModes(const void* fp, const void* sp)
43 {
44  const GLFWvidmode* fm = fp;
45  const GLFWvidmode* sm = sp;
46  const int fbpp = fm->redBits + fm->greenBits + fm->blueBits;
47  const int sbpp = sm->redBits + sm->greenBits + sm->blueBits;
48  const int farea = fm->width * fm->height;
49  const int sarea = sm->width * sm->height;
50 
51  // First sort on color bits per pixel
52  if (fbpp != sbpp)
53  return fbpp - sbpp;
54 
55  // Then sort on screen area
56  if (farea != sarea)
57  return farea - sarea;
58 
59  // Then sort on width
60  if (fm->width != sm->width)
61  return fm->width - sm->width;
62 
63  // Lastly sort on refresh rate
64  return fm->refreshRate - sm->refreshRate;
65 }

67 // Retrieves the available modes for the specified monitor
68 //
69 static GLFWbool refreshVideoModes(_GLFWmonitor* monitor)
70 {
71  int modeCount;
72  GLFWvidmode* modes;
73 
74  if (monitor->modes)
75  return GLFW_TRUE;
76 
77  modes = _glfw.platform.getVideoModes(monitor, &modeCount);
78  if (!modes)
79  return GLFW_FALSE;
80 
81  qsort(modes, modeCount, sizeof(GLFWvidmode), compareVideoModes);
82 
83  _glfw_free(monitor->modes);
84  monitor->modes = modes;
85  monitor->modeCount = modeCount;
86 
87  return GLFW_TRUE;
88 }

95 // Notifies shared code of a monitor connection or disconnection
96 //
97 void _glfwInputMonitor(_GLFWmonitor* monitor, int action, int placement)
98 {
99  if (action == GLFW_CONNECTED)
100  {
101  _glfw.monitorCount++;
102  _glfw.monitors =
103  _glfw_realloc(_glfw.monitors,
104  sizeof(_GLFWmonitor*) * _glfw.monitorCount);
105 
106  if (placement == _GLFW_INSERT_FIRST)
107  {
108  memmove(_glfw.monitors + 1,
109  _glfw.monitors,
110  ((size_t) _glfw.monitorCount - 1) * sizeof(_GLFWmonitor*));
111  _glfw.monitors[0] = monitor;
112  }
113  else
114  _glfw.monitors[_glfw.monitorCount - 1] = monitor;
115  }
116  else if (action == GLFW_DISCONNECTED)
117  {
118  int i;
119  _GLFWwindow* window;
120 
121  for (window = _glfw.windowListHead; window; window = window->next)
122  {
123  if (window->monitor == monitor)
124  {
125  int width, height, xoff, yoff;
126  _glfw.platform.getWindowSize(window, &width, &height);
127  _glfw.platform.setWindowMonitor(window, NULL, 0, 0, width, height, 0);
128  _glfw.platform.getWindowFrameSize(window, &xoff, &yoff, NULL, NULL);
129  _glfw.platform.setWindowPos(window, xoff, yoff);
130  }
131  }
132 
133  for (i = 0; i < _glfw.monitorCount; i++)
134  {
135  if (_glfw.monitors[i] == monitor)
136  {
137  _glfw.monitorCount--;
138  memmove(_glfw.monitors + i,
139  _glfw.monitors + i + 1,
140  ((size_t) _glfw.monitorCount - i) * sizeof(_GLFWmonitor*));
141  break;
142  }
143  }
144  }
145 
146  if (_glfw.callbacks.monitor)
147  _glfw.callbacks.monitor((GLFWmonitor*) monitor, action);
148 
149  if (action == GLFW_DISCONNECTED)
150  _glfwFreeMonitor(monitor);
151 }

216 // Chooses the video mode most closely matching the desired one
217 //
218 const GLFWvidmode* _glfwChooseVideoMode(_GLFWmonitor* monitor,
219  const GLFWvidmode* desired)
220 {
221  int i;
222  unsigned int sizeDiff, leastSizeDiff = UINT_MAX;
223  unsigned int rateDiff, leastRateDiff = UINT_MAX;
224  unsigned int colorDiff, leastColorDiff = UINT_MAX;
225  const GLFWvidmode* current;
226  const GLFWvidmode* closest = NULL;
227 
228  if (!refreshVideoModes(monitor))
229  return NULL;
230 
231  for (i = 0; i < monitor->modeCount; i++)
232  {
233  current = monitor->modes + i;
234 
235  colorDiff = 0;
236 
237  if (desired->redBits != GLFW_DONT_CARE)
238  colorDiff += abs(current->redBits - desired->redBits);
239  if (desired->greenBits != GLFW_DONT_CARE)
240  colorDiff += abs(current->greenBits - desired->greenBits);
241  if (desired->blueBits != GLFW_DONT_CARE)
242  colorDiff += abs(current->blueBits - desired->blueBits);
243 
244  sizeDiff = abs((current->width - desired->width) *
245  (current->width - desired->width) +
246  (current->height - desired->height) *
247  (current->height - desired->height));
248 
249  if (desired->refreshRate != GLFW_DONT_CARE)
250  rateDiff = abs(current->refreshRate - desired->refreshRate);
251  else
252  rateDiff = UINT_MAX - current->refreshRate;
253 
254  if ((colorDiff < leastColorDiff) ||
255  (colorDiff == leastColorDiff && sizeDiff < leastSizeDiff) ||
256  (colorDiff == leastColorDiff && sizeDiff == leastSizeDiff && rateDiff < leastRateDiff))
257  {
258  closest = current;
259  leastSizeDiff = sizeDiff;
260  leastRateDiff = rateDiff;
261  leastColorDiff = colorDiff;
262  }
263  }
264 
265  return closest;
266 }

275 // Splits a color depth into red, green and blue bit depths
276 //
277 void _glfwSplitBPP(int bpp, int* red, int* green, int* blue)
278 {
279  int delta;
280 
281  // We assume that by 32 the user really meant 24
282  if (bpp == 32)
283  bpp = 24;
284 
285  // Convert "bits per pixel" to red, green & blue sizes
286 
287  *red = *green = *blue = bpp / 3;
288  delta = bpp - (*red * 3);
289  if (delta >= 1)
290  *green = *green + 1;
291 
292  if (delta == 2)
293  *red = *red + 1;
294 }

454 GLFWAPI void glfwSetGamma(GLFWmonitor* handle, float gamma)
455 {
456  unsigned int i;
457  unsigned short* values;
458  GLFWgammaramp ramp;
459  const GLFWgammaramp* original;
460  assert(handle != NULL);
461  assert(gamma > 0.f);
462  assert(gamma <= FLT_MAX);
463 
464  _GLFW_REQUIRE_INIT();
465 
466  if (gamma != gamma || gamma <= 0.f || gamma > FLT_MAX)
467  {
468  _glfwInputError(GLFW_INVALID_VALUE, "Invalid gamma value %f", gamma);
469  return;
470  }
471 
472  original = glfwGetGammaRamp(handle);
473  if (!original)
474  return;
475 
476  values = _glfw_calloc(original->size, sizeof(unsigned short));
477 
478  for (i = 0; i < original->size; i++)
479  {
480  float value;
481 
482  // Calculate intensity
483  value = i / (float) (original->size - 1);
484  // Apply gamma curve
485  value = powf(value, 1.f / gamma) * 65535.f + 0.5f;
486  // Clamp to value range
487  value = _glfw_fminf(value, 65535.f);
488 
489  values[i] = (unsigned short) value;
490  }
491 
492  ramp.red = values;
493  ramp.green = values;
494  ramp.blue = values;
495  ramp.size = original->size;
496 
497  glfwSetGammaRamp(handle, &ramp);
498  _glfw_free(values);
499 }
