332 void convert_fastchunk_inplace(fast_chunk *fc)
333 {
334  int i;
335  int num_blocks=0, step=0;
336  unsigned char rot[4096];
337  #ifndef IN_PLACE
338  unsigned char *storage;
339  #endif
340 
341  memset(rot, 0, 4096);
342 
343  for (i=0; i < 16; ++i)
344  num_blocks += fc->blockdata[i] != NULL;
345 
346  #ifndef IN_PLACE
347  storage = malloc(16*16*16*2 * num_blocks);
348  #endif
349 
350  for (i=0; i < 16; ++i) {
351  if (fc->blockdata[i]) {
352  int o=0;
353  unsigned char *bd,*dd,*lt,*sky;
354  unsigned char *out, *outb;
355 
356  // this ordering allows us to determine which data we can safely overwrite for in-place processing
357  bd = fc->blockdata[i];
358  dd = fc->data[i];
359  lt = fc->light[i];
360  sky = fc->skylight[i];
361 
362  #ifdef IN_PLACE
363  out = bd;
364  #else
365  out = storage + 16*16*16*2*step;
366  #endif
367 
368  // bd is written in place, but also reads from dd
369  for (o=0; o < 16*16*16/2; o += 1) {
370  unsigned char v1,v2;
371  unsigned char d = dd[o];
372  v1 = bd[o*2+0];
373  v2 = bd[o*2+1];
374 
375  if (remap[v1])
376  {
377  //unsigned char d = bd[o] & 15;
378  v1 = remap_data[remap[v1]][d&15];
379  rot[o*2+0] = rotate_data[d&3];
380  } else
381  v1 = effective_blocktype[v1];
382 
383  if (remap[v2])
384  {
385  //unsigned char d = bd[o] >> 4;
386  v2 = remap_data[remap[v2]][d>>4];
387  rot[o*2+1] = rotate_data[(d>>4)&3];
388  } else
389  v2 = effective_blocktype[v2];
390 
391  out[o*2+0] = v1;
392  out[o*2+1] = v2;
393  }
394 
395  // this reads from lt & sky
396  #ifndef IN_PLACE
397  outb = out + 16*16*16;
398  ++step;
399  #endif
400 
401  // MC used to write in this order and it makes it possible to compute in-place
402  if (dd < sky && sky < lt) {
403  // @TODO go this path always if !IN_PLACE
404  #ifdef IN_PLACE
405  outb = dd;
406  #endif
407 
408  for (o=0; o < 16*16*16/2; ++o) {
409  int bright;

414
