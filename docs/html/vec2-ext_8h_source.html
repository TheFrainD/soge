218 glm_vec2_complex_div(vec2 a, vec2 b, vec2 dest) {
219  float tr, ti;
220  float const ibnorm2 = 1.0f / (b[0] * b[0] + b[1] * b[1]);
221  tr = ibnorm2 * (a[0] * b[0] + a[1] * b[1]);
222  ti = ibnorm2 * (a[1] * b[0] - a[0] * b[1]);
223  dest[0] = tr;
224  dest[1] = ti;
