128  mat4 res;
129  vec3 v;
130 
131  if (size[0] <= 0.0f || size[1] <= 0.0f)
132  return;
133 
134  /* Translate and scale the picked region to the entire window */
135  v[0] = (vp[2] - 2.0f * (center[0] - vp[0])) / size[0];
136  v[1] = (vp[3] - 2.0f * (center[1] - vp[1])) / size[1];
137  v[2] = 0.0f;
138 
139  glm_translate_make(res, v);
140 
141  v[0] = vp[2] / size[0];
142  v[1] = vp[3] / size[1];
143  v[2] = 1.0f;
144

146 
147  glm_mat4_copy(res, dest);
148 }
