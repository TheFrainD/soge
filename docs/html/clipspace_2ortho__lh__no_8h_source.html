50 glm_ortho_lh_no(float left, float right,
51  float bottom, float top,
52  float nearZ, float farZ,
53  mat4 dest) {
54  float rl, tb, fn;
55 
56  glm_mat4_zero(dest);
57 
58  rl = 1.0f / (right - left);
59  tb = 1.0f / (top - bottom);
60  fn =-1.0f / (farZ - nearZ);
61 
62  dest[0][0] = 2.0f * rl;
63  dest[1][1] = 2.0f * tb;
64  dest[2][2] =-2.0f * fn;
65  dest[3][0] =-(right + left) * rl;
66  dest[3][1] =-(top + bottom) * tb;
67  dest[3][2] = (farZ + nearZ) * fn;
68  dest[3][3] = 1.0f;
69 }
