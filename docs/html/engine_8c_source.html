113b8 engine_run(engine_start start, engine_update update, engine_render render) {
114 if (!start || !update || !render) {
115 VALLY_FATAL("Could not run the engine! Functions must not be NULL!");
116 return FALSE;
117 }
118
119 start();
120
121 while (state.is_runing) {
122 f64 current_time = time_now();
123 f64 delta_time = current_time - state.last_time;
124
125 if (!window_poll_events()) {
126 state.is_runing = FALSE;
127 }
128
129 update(delta_time);
130
131 ecs_animator_update(delta_time);
132
133 renderer_clear_screen();
134 renderer_begin_batch();
135 ecs_spriterenderer_update();
136 render(delta_time);
137 renderer_end_batch();
138 renderer_flush();
139
140 window_swap_buffers();
141
142 state.last_time = current_time;
143 }
144
145 state.is_runing = FALSE;
146
147 // Termiante all systems
148 event_terminate();
149 input_terminate();
150 renderer_terminate();
151 resources_terminate();
152 window_terminate();
153 VALLY_INFO("Shutting down");
154
155 return TRUE;
156}
