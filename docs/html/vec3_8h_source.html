579 void
580 glm_vec3_cross(vec3 a, vec3 b, vec3 dest) {
581  vec3 c;
582  /* (u2.v3 - u3.v2, u3.v1 - u1.v3, u1.v2 - u2.v1) */
583  c[0] = a[1] * b[2] - a[2] * b[1];
584  c[1] = a[2] * b[0] - a[0] * b[2];
585  c[2] = a[0] * b[1] - a[1] * b[0];
586  glm_vec3_copy(c, dest);
587 }

613 glm_vec3_angle(vec3 a, vec3 b) {
614  float norm, dot;
615 
616  /* maybe compiler generate approximation instruction (rcp) */
617  norm = 1.0f / (glm_vec3_norm(a) * glm_vec3_norm(b));
618  dot = glm_vec3_dot(a, b) * norm;
619

635 CGLM_INLINE
636 void
637 glm_vec3_rotate(vec3 v, float angle, vec3 axis) {
638  vec3 v1, v2, k;
639  float c, s;
640 
641  c = cosf(angle);
642  s = sinf(angle);
643 
644  glm_vec3_normalize_to(axis, k);
645 
646  /* Right Hand, Rodrigues' rotation formula:
647  v = v*cos(t) + (kxv)sin(t) + k*(k.v)(1 - cos(t))
648  */
649  glm_vec3_scale(v, c, v1);
650 
651  glm_vec3_cross(k, v, v2);
652  glm_vec3_scale(v2, s, v2);
653 
654  glm_vec3_add(v1, v2, v1);
655 
656  glm_vec3_scale(k, glm_vec3_dot(k, v) * (1.0f - c), v2);
657  glm_vec3_add(v1, v2, v);
658 }
