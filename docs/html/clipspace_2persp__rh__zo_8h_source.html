65 void
66 glm_frustum_rh_zo(float left, float right,
67  float bottom, float top,
68  float nearZ, float farZ,
69  mat4 dest) {
70  float rl, tb, fn, nv;
71 
72  glm_mat4_zero(dest);
73 
74  rl = 1.0f / (right - left);
75  tb = 1.0f / (top - bottom);
76  fn =-1.0f / (farZ - nearZ);
77  nv = 2.0f * nearZ;
78 
79  dest[0][0] = nv * rl;
80  dest[1][1] = nv * tb;
81  dest[2][0] = (right + left) * rl;
82  dest[2][1] = (top + bottom) * tb;
83  dest[2][2] = farZ * fn;
84  dest[2][3] =-1.0f;
85  dest[3][2] = farZ * nearZ * fn;
86 }

99 void
100 glm_perspective_rh_zo(float fovy,
101  float aspect,
102  float nearZ,
103  float farZ,
104  mat4 dest) {
105  float f, fn;
106 
107  glm_mat4_zero(dest);
108 
109  f = 1.0f / tanf(fovy * 0.5f);
110  fn = 1.0f / (nearZ - farZ);
111 
112  dest[0][0] = f / aspect;
113  dest[1][1] = f;
114  dest[2][2] = farZ * fn;
115  dest[2][3] =-1.0f;
116  dest[3][2] = nearZ * farZ * fn;
117 }

160 CGLM_INLINE
161 void
162 glm_persp_move_far_rh_zo(mat4 proj, float deltaFar) {
163  float fn, farZ, nearZ, p22, p32;
164 
165  p22 = proj[2][2];
166  p32 = proj[3][2];
167 
168  nearZ = p32 / p22;
169  farZ = p32 / (p22 + 1.0f) + deltaFar;
170  fn = 1.0f / (nearZ - farZ);
171 
172  proj[2][2] = farZ * fn;
173  proj[3][2] = nearZ * farZ * fn;
174 }

191 glm_persp_decomp_rh_zo(mat4 proj,
192  float * __restrict nearZ, float * __restrict farZ,
193  float * __restrict top, float * __restrict bottom,
194  float * __restrict left, float * __restrict right) {

196  float n_m11, n_m00;
197 
198  m00 = proj[0][0];
199  m11 = proj[1][1];

201  m21 = proj[2][1];
202  m22 = proj[2][2];
203  m32 = proj[3][2];
204 
205  n = m32 / m22;

348 glm_persp_sizes_rh_zo(mat4 proj, float fovy, vec4 dest) {
349  float t, a, nearZ, farZ;
350 
351  t = 2.0f * tanf(fovy * 0.5f);

353 
354  glm_persp_decomp_z_rh_zo(proj, &nearZ, &farZ);
355 
356  dest[1] = t * nearZ;
