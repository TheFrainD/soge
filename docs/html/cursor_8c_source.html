81 static float star(int x, int y, float t)
82 {
83  const float c = 64 / 2.f;
84 
85  const float i = (0.25f * (float) sin(2.f * M_PI * t) + 0.75f);
86  const float k = 64 * 0.046875f * i;
87 
88  const float dist = (float) sqrt((x - c) * (x - c) + (y - c) * (y - c));
89 
90  const float salpha = 1.f - dist / c;
91  const float xalpha = (float) x == c ? c : k / (float) fabs(x - c);
92  const float yalpha = (float) y == c ? c : k / (float) fabs(y - c);
93 
94  return (float) fmax(0.f, fmin(1.f, i * salpha * 0.2f + salpha * xalpha * yalpha));
95 }

97 static GLFWcursor* create_cursor_frame(float t)
98 {
99  int i = 0, x, y;
100  unsigned char buffer[64 * 64 * 4];
101  const GLFWimage image = { 64, 64, buffer };
102 
103  for (y = 0; y < image.width; y++)
104  {
105  for (x = 0; x < image.height; x++)
106  {
107  buffer[i++] = 255;
108  buffer[i++] = 255;
109  buffer[i++] = 255;
110  buffer[i++] = (unsigned char) (255 * star(x, y, t));
111  }
112  }
113 
114  return glfwCreateCursor(&image, image.width / 2, image.height / 2);
115 }

157 static void key_callback(GLFWwindow* window, int key, int scancode, int action, int mods)
158 {
159  if (action != GLFW_PRESS)
160  return;
161 
162  switch (key)
163  {
164  case GLFW_KEY_A:
165  {
166  animate_cursor = !animate_cursor;
167  if (!animate_cursor)
168  glfwSetCursor(window, NULL);
169 
170  break;
171  }
172 
173  case GLFW_KEY_ESCAPE:
174  {
175  if (glfwGetInputMode(window, GLFW_CURSOR) != GLFW_CURSOR_DISABLED)
176  {
177  glfwSetWindowShouldClose(window, GLFW_TRUE);
178  break;
179  }
180 
181  /* FALLTHROUGH */
182  }
183 
184  case GLFW_KEY_N:
185  glfwSetInputMode(window, GLFW_CURSOR, GLFW_CURSOR_NORMAL);
186  glfwGetCursorPos(window, &cursor_x, &cursor_y);
187  printf("(( cursor is normal ))\n");
188  break;
189 
190  case GLFW_KEY_D:
191  glfwSetInputMode(window, GLFW_CURSOR, GLFW_CURSOR_DISABLED);
192  printf("(( cursor is disabled ))\n");
193  break;
194 
195  case GLFW_KEY_H:
196  glfwSetInputMode(window, GLFW_CURSOR, GLFW_CURSOR_HIDDEN);
197  printf("(( cursor is hidden ))\n");
198  break;
199 
200  case GLFW_KEY_R:
201  if (!glfwRawMouseMotionSupported())
202  break;
203 
204  if (glfwGetInputMode(window, GLFW_RAW_MOUSE_MOTION))
205  {
206  glfwSetInputMode(window, GLFW_RAW_MOUSE_MOTION, GLFW_FALSE);
207  printf("(( raw input is disabled ))\n");
208  }
209  else
210  {
211  glfwSetInputMode(window, GLFW_RAW_MOUSE_MOTION, GLFW_TRUE);
212  printf("(( raw input is enabled ))\n");
213  }
214  break;
215 
216  case GLFW_KEY_SPACE:
217  swap_interval = 1 - swap_interval;
218  printf("(( swap interval: %i ))\n", swap_interval);
219  glfwSwapInterval(swap_interval);
220  break;
221 
222  case GLFW_KEY_W:
223  wait_events = !wait_events;
224  printf("(( %sing for events ))\n", wait_events ? "wait" : "poll");
225  break;
226 
227  case GLFW_KEY_T:
228  track_cursor = !track_cursor;
229  if (track_cursor)
230  glfwSetCursor(window, tracking_cursor);
231  else
232  glfwSetCursor(window, NULL);
233 
234  break;
235 
236  case GLFW_KEY_P:
237  {
238  double x, y;
239  glfwGetCursorPos(window, &x, &y);
240 
241  printf("Query before set: %f %f (%+f %+f)\n",
242  x, y, x - cursor_x, y - cursor_y);
243  cursor_x = x;
244  cursor_y = y;
245 
246  glfwSetCursorPos(window, cursor_x, cursor_y);
247  glfwGetCursorPos(window, &x, &y);
248 
249  printf("Query after set: %f %f (%+f %+f)\n",
250  x, y, x - cursor_x, y - cursor_y);
251  cursor_x = x;
252  cursor_y = y;
253  break;
254  }
255 
256  case GLFW_KEY_UP:
257  glfwSetCursorPos(window, 0, 0);
258  glfwGetCursorPos(window, &cursor_x, &cursor_y);
259  break;
260 
261  case GLFW_KEY_DOWN:
262  {
263  int width, height;
264  glfwGetWindowSize(window, &width, &height);
265  glfwSetCursorPos(window, width - 1, height - 1);
266  glfwGetCursorPos(window, &cursor_x, &cursor_y);
267  break;
268  }
269 
270  case GLFW_KEY_0:
271  glfwSetCursor(window, NULL);
272  break;
273 
274  case GLFW_KEY_1:
275  case GLFW_KEY_2:
276  case GLFW_KEY_3:
277  case GLFW_KEY_4:
278  case GLFW_KEY_5:
279  case GLFW_KEY_6:
280  case GLFW_KEY_7:
281  case GLFW_KEY_8:
282  case GLFW_KEY_9:
283  {
284  int index = key - GLFW_KEY_1;
285  if (mods & GLFW_MOD_SHIFT)
286  index += 9;
287 
288  if (index < sizeof(standard_cursors) / sizeof(standard_cursors[0]))
289  glfwSetCursor(window, standard_cursors[index]);
290 
291  break;
292  }
293 
294  case GLFW_KEY_F11:
295  case GLFW_KEY_ENTER:
296  {
297  static int x, y, width, height;
298 
299  if (mods != GLFW_MOD_ALT)
300  return;
301 
302  if (glfwGetWindowMonitor(window))
303  glfwSetWindowMonitor(window, NULL, x, y, width, height, 0);
304  else
305  {
306  GLFWmonitor* monitor = glfwGetPrimaryMonitor();
307  const GLFWvidmode* mode = glfwGetVideoMode(monitor);
308  glfwGetWindowPos(window, &x, &y);
309  glfwGetWindowSize(window, &width, &height);
310  glfwSetWindowMonitor(window, monitor,
311  0, 0, mode->width, mode->height,
312  mode->refreshRate);
313  }
314 
315  glfwGetCursorPos(window, &cursor_x, &cursor_y);
316  break;
317  }
318  }

321 int main(void)
322 {
323  int i;
324  GLFWwindow* window;
325  GLFWcursor* star_cursors[CURSOR_FRAME_COUNT];
326  GLFWcursor* current_frame = NULL;
327  GLuint vertex_buffer, vertex_shader, fragment_shader, program;
328  GLint mvp_location, vpos_location;
329 
330  glfwSetErrorCallback(error_callback);
331 
332  if (!glfwInit())
333  exit(EXIT_FAILURE);
334 
335  tracking_cursor = create_tracking_cursor();
336  if (!tracking_cursor)
337  {
338  glfwTerminate();
339  exit(EXIT_FAILURE);
340  }
341 
342  for (i = 0; i < CURSOR_FRAME_COUNT; i++)
343  {
344  star_cursors[i] = create_cursor_frame(i / (float) CURSOR_FRAME_COUNT);
345  if (!star_cursors[i])
346  {
347  glfwTerminate();
348  exit(EXIT_FAILURE);
349  }
350  }
351 
352  for (i = 0; i < sizeof(standard_cursors) / sizeof(standard_cursors[0]); i++)
353  {
354  const int shapes[] = {
355  GLFW_ARROW_CURSOR,
356  GLFW_IBEAM_CURSOR,
357  GLFW_CROSSHAIR_CURSOR,
358  GLFW_POINTING_HAND_CURSOR,
359  GLFW_RESIZE_EW_CURSOR,
360  GLFW_RESIZE_NS_CURSOR,
361  GLFW_RESIZE_NWSE_CURSOR,
362  GLFW_RESIZE_NESW_CURSOR,
363  GLFW_RESIZE_ALL_CURSOR,
364  GLFW_NOT_ALLOWED_CURSOR
365  };
366 
367  standard_cursors[i] = glfwCreateStandardCursor(shapes[i]);
368  }
369 
370  glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, 2);
371  glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, 0);
372 
373  window = glfwCreateWindow(640, 480, "Cursor Test", NULL, NULL);
374  if (!window)
375  {
376  glfwTerminate();
377  exit(EXIT_FAILURE);
378  }
379 
380  glfwMakeContextCurrent(window);
381  gladLoadGL(glfwGetProcAddress);
382 
383  glGenBuffers(1, &vertex_buffer);
384  glBindBuffer(GL_ARRAY_BUFFER, vertex_buffer);
385 
386  vertex_shader = glCreateShader(GL_VERTEX_SHADER);
387  glShaderSource(vertex_shader, 1, &vertex_shader_text, NULL);
388  glCompileShader(vertex_shader);
389 
390  fragment_shader = glCreateShader(GL_FRAGMENT_SHADER);
391  glShaderSource(fragment_shader, 1, &fragment_shader_text, NULL);
392  glCompileShader(fragment_shader);
393 
394  program = glCreateProgram();
395  glAttachShader(program, vertex_shader);
396  glAttachShader(program, fragment_shader);
397  glLinkProgram(program);
398 
399  mvp_location = glGetUniformLocation(program, "MVP");
400  vpos_location = glGetAttribLocation(program, "vPos");
401 
402  glEnableVertexAttribArray(vpos_location);
403  glVertexAttribPointer(vpos_location, 2, GL_FLOAT, GL_FALSE,
404  sizeof(vec2), (void*) 0);
405  glUseProgram(program);
406 
407  glfwGetCursorPos(window, &cursor_x, &cursor_y);
408  printf("Cursor position: %f %f\n", cursor_x, cursor_y);
409 
410  glfwSetCursorPosCallback(window, cursor_position_callback);
411  glfwSetKeyCallback(window, key_callback);
412 
413  while (!glfwWindowShouldClose(window))
414  {
415  glClear(GL_COLOR_BUFFER_BIT);
416 
417  if (track_cursor)
418  {
419  int wnd_width, wnd_height, fb_width, fb_height;
420  float scale;
421  vec2 vertices[4];
422  mat4x4 mvp;
423 
424  glfwGetWindowSize(window, &wnd_width, &wnd_height);
425  glfwGetFramebufferSize(window, &fb_width, &fb_height);
426 
427  glViewport(0, 0, fb_width, fb_height);
428 
429  scale = (float) fb_width / (float) wnd_width;
430  vertices[0][0] = 0.5f;
431  vertices[0][1] = (float) (fb_height - floor(cursor_y * scale) - 1.f + 0.5f);
432  vertices[1][0] = (float) fb_width + 0.5f;
433  vertices[1][1] = (float) (fb_height - floor(cursor_y * scale) - 1.f + 0.5f);
434  vertices[2][0] = (float) floor(cursor_x * scale) + 0.5f;
435  vertices[2][1] = 0.5f;
436  vertices[3][0] = (float) floor(cursor_x * scale) + 0.5f;
437  vertices[3][1] = (float) fb_height + 0.5f;
438 
439  glBufferData(GL_ARRAY_BUFFER,
440  sizeof(vertices),
441  vertices,
442  GL_STREAM_DRAW);
443 
444  mat4x4_ortho(mvp, 0.f, (float) fb_width, 0.f, (float) fb_height, 0.f, 1.f);
445  glUniformMatrix4fv(mvp_location, 1, GL_FALSE, (const GLfloat*) mvp);
446 
447  glDrawArrays(GL_LINES, 0, 4);
448  }
449 
450  glfwSwapBuffers(window);
451 
452  if (animate_cursor)
453  {
454  const int i = (int) (glfwGetTime() * 30.0) % CURSOR_FRAME_COUNT;
455  if (current_frame != star_cursors[i])
456  {
457  glfwSetCursor(window, star_cursors[i]);
458  current_frame = star_cursors[i];
459  }
460  }
461  else
462  current_frame = NULL;
463 
464  if (wait_events)
465  {
466  if (animate_cursor)
467  glfwWaitEventsTimeout(1.0 / 30.0);
468  else
469  glfwWaitEvents();
470  }
471  else
472  glfwPollEvents();
473 
474  // Workaround for an issue with msvcrt and mintty
475  fflush(stdout);
476  }
477 
478  glfwDestroyWindow(window);
479 
480  for (i = 0; i < CURSOR_FRAME_COUNT; i++)
481  glfwDestroyCursor(star_cursors[i]);
482 
483  for (i = 0; i < sizeof(standard_cursors) / sizeof(standard_cursors[0]); i++)
484  glfwDestroyCursor(standard_cursors[i]);
485 
486  glfwTerminate();
487  exit(EXIT_SUCCESS);
488 }
