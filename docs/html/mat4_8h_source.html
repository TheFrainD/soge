426 CGLM_INLINE
427 void
428 glm_mat4_quat(mat4 m, versor dest) {
429  float trace, r, rinv;
430 
431  /* it seems using like m12 instead of m[1][2] causes extra instructions */
432 
433  trace = m[0][0] + m[1][1] + m[2][2];
434  if (trace >= 0.0f) {
435  r = sqrtf(1.0f + trace);
436  rinv = 0.5f / r;
437 
438  dest[0] = rinv * (m[1][2] - m[2][1]);
439  dest[1] = rinv * (m[2][0] - m[0][2]);
440  dest[2] = rinv * (m[0][1] - m[1][0]);
441  dest[3] = r * 0.5f;
442  } else if (m[0][0] >= m[1][1] && m[0][0] >= m[2][2]) {
443  r = sqrtf(1.0f - m[1][1] - m[2][2] + m[0][0]);
444  rinv = 0.5f / r;
445 
446  dest[0] = r * 0.5f;
447  dest[1] = rinv * (m[0][1] + m[1][0]);
448  dest[2] = rinv * (m[0][2] + m[2][0]);
449  dest[3] = rinv * (m[1][2] - m[2][1]);
450  } else if (m[1][1] >= m[2][2]) {
451  r = sqrtf(1.0f - m[0][0] - m[2][2] + m[1][1]);
452  rinv = 0.5f / r;
453 
454  dest[0] = rinv * (m[0][1] + m[1][0]);
455  dest[1] = r * 0.5f;
456  dest[2] = rinv * (m[1][2] + m[2][1]);
457  dest[3] = rinv * (m[2][0] - m[0][2]);
458  } else {
459  r = sqrtf(1.0f - m[0][0] - m[1][1] + m[2][2]);
460  rinv = 0.5f / r;
461 
462  dest[0] = rinv * (m[0][2] + m[2][0]);
463  dest[1] = rinv * (m[1][2] + m[2][1]);
464  dest[2] = r * 0.5f;
465  dest[3] = rinv * (m[0][1] - m[1][0]);
466  }
467 }

581 CGLM_INLINE
582 float
583 glm_mat4_det(mat4 mat) {
584 #if defined( __SSE__ ) || defined( __SSE2__ )
585  return glm_mat4_det_sse2(mat);
586 #elif defined(CGLM_NEON_FP)
587  return glm_mat4_det_neon(mat);
588 #else
589  /* [square] det(A) = det(At) */
590  float t[6];
591  float a = mat[0][0], b = mat[0][1], c = mat[0][2], d = mat[0][3],
592  e = mat[1][0], f = mat[1][1], g = mat[1][2], h = mat[1][3],
593  i = mat[2][0], j = mat[2][1], k = mat[2][2], l = mat[2][3],
594  m = mat[3][0], n = mat[3][1], o = mat[3][2], p = mat[3][3];
595 
596  t[0] = k * p - o * l;
597  t[1] = j * p - n * l;
598  t[2] = j * o - n * k;
599  t[3] = i * p - m * l;
600  t[4] = i * o - m * k;
601  t[5] = i * n - m * j;
602 
603  return a * (f * t[0] - g * t[1] + h * t[2])
604  - b * (e * t[0] - g * t[3] + h * t[4])
605  + c * (e * t[1] - f * t[3] + h * t[5])
606  - d * (e * t[2] - f * t[4] + g * t[5]);
607 #endif
608 }

618 glm_mat4_inv(mat4 mat, mat4 dest) {
619 #if defined( __SSE__ ) || defined( __SSE2__ )
620  glm_mat4_inv_sse2(mat, dest);
621 #elif defined(CGLM_NEON_FP)
622  glm_mat4_inv_neon(mat, dest);
623 #else
624  float t[6];
625  float det;
626  float a = mat[0][0], b = mat[0][1], c = mat[0][2], d = mat[0][3],
627  e = mat[1][0], f = mat[1][1], g = mat[1][2], h = mat[1][3],
628  i = mat[2][0], j = mat[2][1], k = mat[2][2], l = mat[2][3],
629  m = mat[3][0], n = mat[3][1], o = mat[3][2], p = mat[3][3];
630 
631  t[0] = k * p - o * l; t[1] = j * p - n * l; t[2] = j * o - n * k;
632  t[3] = i * p - m * l; t[4] = i * o - m * k; t[5] = i * n - m * j;
633 
634  dest[0][0] = f * t[0] - g * t[1] + h * t[2];
635  dest[1][0] =-(e * t[0] - g * t[3] + h * t[4]);
636  dest[2][0] = e * t[1] - f * t[3] + h * t[5];
637  dest[3][0] =-(e * t[2] - f * t[4] + g * t[5]);
638 
639  dest[0][1] =-(b * t[0] - c * t[1] + d * t[2]);
640  dest[1][1] = a * t[0] - c * t[3] + d * t[4];
641  dest[2][1] =-(a * t[1] - b * t[3] + d * t[5]);
