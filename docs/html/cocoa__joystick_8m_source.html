76 // Comparison function for matching the SDL element order
77 //
78 static CFComparisonResult compareElements(const void* fp,
79  const void* sp,
80  void* user)
81 {
82  const _GLFWjoyelementNS* fe = fp;
83  const _GLFWjoyelementNS* se = sp;
84  if (fe->usage < se->usage)
85  return kCFCompareLessThan;
86  if (fe->usage > se->usage)
87  return kCFCompareGreaterThan;
88  if (fe->index < se->index)
89  return kCFCompareLessThan;
90  if (fe->index > se->index)
91  return kCFCompareGreaterThan;
92  return kCFCompareEqualTo;
93 }

118 // Callback for user-initiated joystick addition
119 //
120 static void matchCallback(void* context,
121  IOReturn result,
122  void* sender,
123  IOHIDDeviceRef device)
124 {
125  int jid;
126  char name[256];
127  char guid[33];
128  CFTypeRef property;
129  uint32_t vendor = 0, product = 0, version = 0;
130  _GLFWjoystick* js;
131  CFMutableArrayRef axes, buttons, hats;
132 
133  for (jid = 0; jid <= GLFW_JOYSTICK_LAST; jid++)
134  {
135  if (_glfw.joysticks[jid].ns.device == device)
136  return;
137  }
138 
139  axes = CFArrayCreateMutable(NULL, 0, NULL);
140  buttons = CFArrayCreateMutable(NULL, 0, NULL);
141  hats = CFArrayCreateMutable(NULL, 0, NULL);
142 
143  property = IOHIDDeviceGetProperty(device, CFSTR(kIOHIDProductKey));
144  if (property)
145  {
146  CFStringGetCString(property,
147  name,
148  sizeof(name),
149  kCFStringEncodingUTF8);
150  }
151  else
152  strncpy(name, "Unknown", sizeof(name));
153 
154  property = IOHIDDeviceGetProperty(device, CFSTR(kIOHIDVendorIDKey));
155  if (property)
156  CFNumberGetValue(property, kCFNumberSInt32Type, &vendor);
157 
158  property = IOHIDDeviceGetProperty(device, CFSTR(kIOHIDProductIDKey));
159  if (property)
160  CFNumberGetValue(property, kCFNumberSInt32Type, &product);
161 
162  property = IOHIDDeviceGetProperty(device, CFSTR(kIOHIDVersionNumberKey));
163  if (property)
164  CFNumberGetValue(property, kCFNumberSInt32Type, &version);
165 
166  // Generate a joystick GUID that matches the SDL 2.0.5+ one
167  if (vendor && product)
168  {
169  sprintf(guid, "03000000%02x%02x0000%02x%02x0000%02x%02x0000",
170  (uint8_t) vendor, (uint8_t) (vendor >> 8),
171  (uint8_t) product, (uint8_t) (product >> 8),
172  (uint8_t) version, (uint8_t) (version >> 8));
173  }
174  else
175  {
176  sprintf(guid, "05000000%02x%02x%02x%02x%02x%02x%02x%02x%02x%02x%02x00",
177  name[0], name[1], name[2], name[3],
178  name[4], name[5], name[6], name[7],
179  name[8], name[9], name[10]);
180  }
181 
182  CFArrayRef elements =
183  IOHIDDeviceCopyMatchingElements(device, NULL, kIOHIDOptionsTypeNone);
184 
185  for (CFIndex i = 0; i < CFArrayGetCount(elements); i++)
186  {
187  IOHIDElementRef native = (IOHIDElementRef)
188  CFArrayGetValueAtIndex(elements, i);
189  if (CFGetTypeID(native) != IOHIDElementGetTypeID())
190  continue;
191 
192  const IOHIDElementType type = IOHIDElementGetType(native);
193  if ((type != kIOHIDElementTypeInput_Axis) &&
194  (type != kIOHIDElementTypeInput_Button) &&
195  (type != kIOHIDElementTypeInput_Misc))
196  {
197  continue;
198  }
199 
200  CFMutableArrayRef target = NULL;
201 
202  const uint32_t usage = IOHIDElementGetUsage(native);
203  const uint32_t page = IOHIDElementGetUsagePage(native);
204  if (page == kHIDPage_GenericDesktop)
205  {
206  switch (usage)
207  {
208  case kHIDUsage_GD_X:
209  case kHIDUsage_GD_Y:
210  case kHIDUsage_GD_Z:
211  case kHIDUsage_GD_Rx:
212  case kHIDUsage_GD_Ry:
213  case kHIDUsage_GD_Rz:
214  case kHIDUsage_GD_Slider:
215  case kHIDUsage_GD_Dial:
216  case kHIDUsage_GD_Wheel:
217  target = axes;
218  break;
219  case kHIDUsage_GD_Hatswitch:
220  target = hats;
221  break;
222  case kHIDUsage_GD_DPadUp:
223  case kHIDUsage_GD_DPadRight:
224  case kHIDUsage_GD_DPadDown:
225  case kHIDUsage_GD_DPadLeft:
226  case kHIDUsage_GD_SystemMainMenu:
227  case kHIDUsage_GD_Select:
228  case kHIDUsage_GD_Start:
229  target = buttons;
230  break;
231  }
232  }
233  else if (page == kHIDPage_Simulation)
234  {
235  switch (usage)
236  {
237  case kHIDUsage_Sim_Accelerator:
238  case kHIDUsage_Sim_Brake:
239  case kHIDUsage_Sim_Throttle:
240  case kHIDUsage_Sim_Rudder:
241  case kHIDUsage_Sim_Steering:
242  target = axes;
243  break;
244  }
245  }
246  else if (page == kHIDPage_Button || page == kHIDPage_Consumer)
247  target = buttons;
248 
249  if (target)
250  {
251  _GLFWjoyelementNS* element = _glfw_calloc(1, sizeof(_GLFWjoyelementNS));
252  element->native = native;
253  element->usage = usage;
254  element->index = (int) CFArrayGetCount(target);
255  element->minimum = IOHIDElementGetLogicalMin(native);
256  element->maximum = IOHIDElementGetLogicalMax(native);
257  CFArrayAppendValue(target, element);
258  }
259  }
260 
261  CFRelease(elements);
262 
263  CFArraySortValues(axes, CFRangeMake(0, CFArrayGetCount(axes)),
264  compareElements, NULL);
265  CFArraySortValues(buttons, CFRangeMake(0, CFArrayGetCount(buttons)),
266  compareElements, NULL);
267  CFArraySortValues(hats, CFRangeMake(0, CFArrayGetCount(hats)),
268  compareElements, NULL);
269 
270  js = _glfwAllocJoystick(name, guid,
271  (int) CFArrayGetCount(axes),
272  (int) CFArrayGetCount(buttons),
273  (int) CFArrayGetCount(hats));
274 
275  js->ns.device = device;
276  js->ns.axes = axes;
277  js->ns.buttons = buttons;
278  js->ns.hats = hats;
279 
280  _glfwInputJoystick(js, GLFW_CONNECTED);
281 }

305 GLFWbool _glfwInitJoysticksCocoa(void)
306 {
307  CFMutableArrayRef matching;
308  const long usages[] =
309  {
310  kHIDUsage_GD_Joystick,
311  kHIDUsage_GD_GamePad,
312  kHIDUsage_GD_MultiAxisController
313  };
314 
315  _glfw.ns.hidManager = IOHIDManagerCreate(kCFAllocatorDefault,
316  kIOHIDOptionsTypeNone);
317 
318  matching = CFArrayCreateMutable(kCFAllocatorDefault,
319  0,
320  &kCFTypeArrayCallBacks);
321  if (!matching)
322  {
323  _glfwInputError(GLFW_PLATFORM_ERROR, "Cocoa: Failed to create array");
324  return GLFW_FALSE;
325  }
326 
327  for (size_t i = 0; i < sizeof(usages) / sizeof(long); i++)
328  {
329  const long page = kHIDPage_GenericDesktop;
330 
331  CFMutableDictionaryRef dict =
332  CFDictionaryCreateMutable(kCFAllocatorDefault,
333  0,
334  &kCFTypeDictionaryKeyCallBacks,
335  &kCFTypeDictionaryValueCallBacks);
336  if (!dict)
337  continue;
338 
339  CFNumberRef pageRef = CFNumberCreate(kCFAllocatorDefault,
340  kCFNumberLongType,
341  &page);
342  CFNumberRef usageRef = CFNumberCreate(kCFAllocatorDefault,
343  kCFNumberLongType,
344  &usages[i]);
345  if (pageRef && usageRef)
346  {
347  CFDictionarySetValue(dict,
348  CFSTR(kIOHIDDeviceUsagePageKey),
349  pageRef);
350  CFDictionarySetValue(dict,
351  CFSTR(kIOHIDDeviceUsageKey),
352  usageRef);
353  CFArrayAppendValue(matching, dict);
354  }
355 
356  if (pageRef)
357  CFRelease(pageRef);
358  if (usageRef)
359  CFRelease(usageRef);
360 
361  CFRelease(dict);
362  }
363 
364  IOHIDManagerSetDeviceMatchingMultiple(_glfw.ns.hidManager, matching);
365  CFRelease(matching);
366 
367  IOHIDManagerRegisterDeviceMatchingCallback(_glfw.ns.hidManager,
368  &matchCallback, NULL);
369  IOHIDManagerRegisterDeviceRemovalCallback(_glfw.ns.hidManager,
370  &removeCallback, NULL);
371  IOHIDManagerScheduleWithRunLoop(_glfw.ns.hidManager,
372  CFRunLoopGetMain(),
373  kCFRunLoopDefaultMode);
374  IOHIDManagerOpen(_glfw.ns.hidManager, kIOHIDOptionsTypeNone);
375 
376  // Execute the run loop once in order to register any initially-attached
377  // joysticks
378  CFRunLoopRunInMode(kCFRunLoopDefaultMode, 0, false);
379  return GLFW_TRUE;
380 }

395 int _glfwPollJoystickCocoa(_GLFWjoystick* js, int mode)
396 {
397  if (mode & _GLFW_POLL_AXES)
398  {
399  for (CFIndex i = 0; i < CFArrayGetCount(js->ns.axes); i++)
400  {
401  _GLFWjoyelementNS* axis = (_GLFWjoyelementNS*)
402  CFArrayGetValueAtIndex(js->ns.axes, i);
403 
404  const long raw = getElementValue(js, axis);
405  // Perform auto calibration
406  if (raw < axis->minimum)
407  axis->minimum = raw;
408  if (raw > axis->maximum)
409  axis->maximum = raw;
410 
411  const long size = axis->maximum - axis->minimum;
412  if (size == 0)
413  _glfwInputJoystickAxis(js, (int) i, 0.f);
414  else
415  {
416  const float value = (2.f * (raw - axis->minimum) / size) - 1.f;
417  _glfwInputJoystickAxis(js, (int) i, value);
418  }
419  }
420  }
421 
422  if (mode & _GLFW_POLL_BUTTONS)
423  {
424  for (CFIndex i = 0; i < CFArrayGetCount(js->ns.buttons); i++)
425  {
426  _GLFWjoyelementNS* button = (_GLFWjoyelementNS*)
427  CFArrayGetValueAtIndex(js->ns.buttons, i);
428  const char value = getElementValue(js, button) - button->minimum;
429  const int state = (value > 0) ? GLFW_PRESS : GLFW_RELEASE;
430  _glfwInputJoystickButton(js, (int) i, state);
431  }
432 
433  for (CFIndex i = 0; i < CFArrayGetCount(js->ns.hats); i++)
434  {
435  const int states[9] =
436  {
437  GLFW_HAT_UP,
438  GLFW_HAT_RIGHT_UP,
439  GLFW_HAT_RIGHT,
440  GLFW_HAT_RIGHT_DOWN,
441  GLFW_HAT_DOWN,
442  GLFW_HAT_LEFT_DOWN,
443  GLFW_HAT_LEFT,
444  GLFW_HAT_LEFT_UP,
445  GLFW_HAT_CENTERED
446  };
447 
448  _GLFWjoyelementNS* hat = (_GLFWjoyelementNS*)
449  CFArrayGetValueAtIndex(js->ns.hats, i);
450  long state = getElementValue(js, hat) - hat->minimum;
451  if (state < 0 || state > 8)
452  state = 8;
453 
454  _glfwInputJoystickHat(js, (int) i, states[state]);
455  }
456  }
457 
458  return js->present;
459 }
