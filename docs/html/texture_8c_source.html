34 texture *texture_create(image *image) {
35  texture *_texture = (texture *)malloc(sizeof(texture));
36  if (!_texture) {
37  VALLY_ERROR("Could not create texture");
38  }
39 
40  // Generate OpenGL Texture
41  glGenTextures(1, &_texture->id);
42  glBindTexture(GL_TEXTURE_2D, _texture->id);
43 
44  // Configure texture parameters
45  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
46  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
47  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_NEAREST);
48  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_NEAREST);
49 
50  _texture->width = image->width;
51  _texture->height = image->height; // Binding image to the texture
52  glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA, _texture->width, _texture->height, 0, GL_RGBA, GL_UNSIGNED_BYTE, image->buffer);
53 
54  glBindTexture(GL_TEXTURE_2D, 0);
55 
56  resources_add(_texture, texture_destroy);
57 
58  return _texture;
59 }
