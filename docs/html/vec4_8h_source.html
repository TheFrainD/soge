698 void
699 glm_vec4_normalize_to(vec4 v, vec4 dest) {
700 #if defined( __SSE__ ) || defined( __SSE2__ )
701  __m128 xdot, x0;
702  float dot;
703 
704  x0 = glmm_load(v);
705  xdot = glmm_vdot(x0, x0);
706  dot = _mm_cvtss_f32(xdot);
707 
708  if (dot == 0.0f) {
709  glmm_store(dest, _mm_setzero_ps());
710  return;
711  }
712 
713  glmm_store(dest, _mm_div_ps(x0, _mm_sqrt_ps(xdot)));
714 #else
715  float norm;
716 
717  norm = glm_vec4_norm(v);
718 
719  if (norm == 0.0f) {
720  glm_vec4_zero(dest);
721  return;
722  }
723 
724  glm_vec4_scale(v, 1.0f / norm, dest);
725 #endif
