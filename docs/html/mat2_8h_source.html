256 CGLM_INLINE
257 void
258 glm_mat2_inv(mat2 mat, mat2 dest) {
259  float det;
260  float a = mat[0][0], b = mat[0][1],
261  c = mat[1][0], d = mat[1][1];
262 
263  det = 1.0f / (a * d - b * c);
264 
265  dest[0][0] = d * det;
266  dest[0][1] = -b * det;
267  dest[1][0] = -c * det;
268  dest[1][1] = a * det;
269 }
