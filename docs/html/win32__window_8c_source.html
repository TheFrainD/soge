77 //
78 static const GLFWimage* chooseImage(int count, const GLFWimage* images,
79  int width, int height)
80 {
81  int i, leastDiff = INT_MAX;
82  const GLFWimage* closest = NULL;
83 
84  for (i = 0; i < count; i++)
85  {
86  const int currDiff = abs(images[i].width * images[i].height -
87  width * height);
88  if (currDiff < leastDiff)
89  {
90  closest = images + i;
91  leastDiff = currDiff;
92  }
93  }

98 // Creates an RGBA icon or cursor
99 //
100 static HICON createIcon(const GLFWimage* image,
101  int xhot, int yhot, GLFWbool icon)
102 {
103  int i;
104  HDC dc;
105  HICON handle;
106  HBITMAP color, mask;
107  BITMAPV5HEADER bi;
108  ICONINFO ii;
109  unsigned char* target = NULL;
110  unsigned char* source = image->pixels;
111 
112  ZeroMemory(&bi, sizeof(bi));
113  bi.bV5Size = sizeof(bi);
114  bi.bV5Width = image->width;
115  bi.bV5Height = -image->height;
116  bi.bV5Planes = 1;
117  bi.bV5BitCount = 32;
118  bi.bV5Compression = BI_BITFIELDS;
119  bi.bV5RedMask = 0x00ff0000;
120  bi.bV5GreenMask = 0x0000ff00;
121  bi.bV5BlueMask = 0x000000ff;
122  bi.bV5AlphaMask = 0xff000000;
123 
124  dc = GetDC(NULL);
125  color = CreateDIBSection(dc,
126  (BITMAPINFO*) &bi,
127  DIB_RGB_COLORS,
128  (void**) &target,
129  NULL,
130  (DWORD) 0);
131  ReleaseDC(NULL, dc);
132 
133  if (!color)
134  {
135  _glfwInputErrorWin32(GLFW_PLATFORM_ERROR,
136  "Win32: Failed to create RGBA bitmap");
137  return NULL;
138  }
139 
140  mask = CreateBitmap(image->width, image->height, 1, 1, NULL);
141  if (!mask)
142  {
143  _glfwInputErrorWin32(GLFW_PLATFORM_ERROR,
144  "Win32: Failed to create mask bitmap");
145  DeleteObject(color);
146  return NULL;
147  }
148 
149  for (i = 0; i < image->width * image->height; i++)
150  {
151  target[0] = source[2];
152  target[1] = source[1];
153  target[2] = source[0];
154  target[3] = source[3];
155  target += 4;
156  source += 4;
157  }
158 
159  ZeroMemory(&ii, sizeof(ii));
160  ii.fIcon = icon;
161  ii.xHotspot = xhot;
162  ii.yHotspot = yhot;
163  ii.hbmMask = mask;
164  ii.hbmColor = color;
165 
166  handle = CreateIconIndirect(&ii);
167 
168  DeleteObject(color);
169  DeleteObject(mask);
170 
171  if (!handle)
172  {
173  if (icon)
174  {
175  _glfwInputErrorWin32(GLFW_PLATFORM_ERROR,
176  "Win32: Failed to create icon");
177  }
178  else
179  {
180  _glfwInputErrorWin32(GLFW_PLATFORM_ERROR,
181  "Win32: Failed to create cursor");
182  }
183  }
184 
185  return handle;
186 }

206 // Enforce the content area aspect ratio based on which edge is being dragged
207 //
208 static void applyAspectRatio(_GLFWwindow* window, int edge, RECT* area)
209 {
210  int xoff, yoff;
211  UINT dpi = USER_DEFAULT_SCREEN_DPI;
212  const float ratio = (float) window->numer / (float) window->denom;
213 
214  if (_glfwIsWindows10AnniversaryUpdateOrGreaterWin32())
215  dpi = GetDpiForWindow(window->win32.handle);
216 
217  getFullWindowSize(getWindowStyle(window), getWindowExStyle(window),
218  0, 0, &xoff, &yoff, dpi);
219 
220  if (edge == WMSZ_LEFT || edge == WMSZ_BOTTOMLEFT ||
221  edge == WMSZ_RIGHT || edge == WMSZ_BOTTOMRIGHT)
222  {
223  area->bottom = area->top + yoff +
224  (int) ((area->right - area->left - xoff) / ratio);
225  }
226  else if (edge == WMSZ_TOPLEFT || edge == WMSZ_TOPRIGHT)
227  {
228  area->top = area->bottom - yoff -
229  (int) ((area->right - area->left - xoff) / ratio);
230  }
231  else if (edge == WMSZ_TOP || edge == WMSZ_BOTTOM)
232  {
233  area->right = area->left + xoff +
234  (int) ((area->bottom - area->top - yoff) * ratio);
235  }
236 }

375 // Update window framebuffer transparency
376 //
377 static void updateFramebufferTransparency(const _GLFWwindow* window)
378 {
379  BOOL composition, opaque;
380  DWORD color;
381 
382  if (!IsWindowsVistaOrGreater())
383  return;
384 
385  if (FAILED(DwmIsCompositionEnabled(&composition)) || !composition)
386  return;
387 
388  if (IsWindows8OrGreater() ||
389  (SUCCEEDED(DwmGetColorizationColor(&color, &opaque)) && !opaque))
390  {
391  HRGN region = CreateRectRgn(0, 0, -1, -1);
392  DWM_BLURBEHIND bb = {0};
393  bb.dwFlags = DWM_BB_ENABLE | DWM_BB_BLURREGION;
394  bb.hRgnBlur = region;
395  bb.fEnable = TRUE;
396 
397  DwmEnableBlurBehindWindow(window->win32.handle, &bb);
398  DeleteObject(region);
399  }
400  else
401  {
402  // HACK: Disable framebuffer transparency on Windows 7 when the
403  // colorization color is opaque, because otherwise the window
404  // contents is blended additively with the previous frame instead
405  // of replacing it
406  DWM_BLURBEHIND bb = {0};
407  bb.dwFlags = DWM_BB_ENABLE;
408  DwmEnableBlurBehindWindow(window->win32.handle, &bb);
409  }

412 // Retrieves and translates modifier keys
413 //
414 static int getKeyMods(void)
415 {
416  int mods = 0;
417 
418  if (GetKeyState(VK_SHIFT) & 0x8000)
419  mods |= GLFW_MOD_SHIFT;
420  if (GetKeyState(VK_CONTROL) & 0x8000)
421  mods |= GLFW_MOD_CONTROL;
422  if (GetKeyState(VK_MENU) & 0x8000)
423  mods |= GLFW_MOD_ALT;
424  if ((GetKeyState(VK_LWIN) | GetKeyState(VK_RWIN)) & 0x8000)
425  mods |= GLFW_MOD_SUPER;
426  if (GetKeyState(VK_CAPITAL) & 1)
427  mods |= GLFW_MOD_CAPS_LOCK;
428  if (GetKeyState(VK_NUMLOCK) & 1)
429  mods |= GLFW_MOD_NUM_LOCK;
430 
431  return mods;
432 }

487 // Window callback function (handles window messages)
488 //
489 static LRESULT CALLBACK windowProc(HWND hWnd, UINT uMsg,
490  WPARAM wParam, LPARAM lParam)
491 {
492  _GLFWwindow* window = GetPropW(hWnd, L"GLFW");
493  if (!window)
494  {
495  // This is the message handling for the hidden helper window
496  // and for a regular window during its initial creation
497 
498  switch (uMsg)
499  {
500  case WM_NCCREATE:
501  {
502  if (_glfwIsWindows10AnniversaryUpdateOrGreaterWin32())
503  {
504  const CREATESTRUCTW* cs = (const CREATESTRUCTW*) lParam;
505  const _GLFWwndconfig* wndconfig = cs->lpCreateParams;
506 
507  // On per-monitor DPI aware V1 systems, only enable
508  // non-client scaling for windows that scale the client area
509  // We need WM_GETDPISCALEDSIZE from V2 to keep the client
510  // area static when the non-client area is scaled
511  if (wndconfig && wndconfig->scaleToMonitor)
512  EnableNonClientDpiScaling(hWnd);
513  }
514 
515  break;
516  }
517 
518  case WM_DISPLAYCHANGE:
519  _glfwPollMonitorsWin32();
520  break;
521 
522  case WM_DEVICECHANGE:
523  {
524  if (!_glfw.joysticksInitialized)
525  break;
526 
527  if (wParam == DBT_DEVICEARRIVAL)
528  {
529  DEV_BROADCAST_HDR* dbh = (DEV_BROADCAST_HDR*) lParam;
530  if (dbh && dbh->dbch_devicetype == DBT_DEVTYP_DEVICEINTERFACE)
531  _glfwDetectJoystickConnectionWin32();
532  }
533  else if (wParam == DBT_DEVICEREMOVECOMPLETE)
534  {
535  DEV_BROADCAST_HDR* dbh = (DEV_BROADCAST_HDR*) lParam;
536  if (dbh && dbh->dbch_devicetype == DBT_DEVTYP_DEVICEINTERFACE)
537  _glfwDetectJoystickDisconnectionWin32();
538  }
539 
540  break;
541  }
542  }
543 
544  return DefWindowProcW(hWnd, uMsg, wParam, lParam);
545  }
546 
547  switch (uMsg)
548  {
549  case WM_MOUSEACTIVATE:
550  {
551  // HACK: Postpone cursor disabling when the window was activated by
552  // clicking a caption button
553  if (HIWORD(lParam) == WM_LBUTTONDOWN)
554  {
555  if (LOWORD(lParam) != HTCLIENT)
556  window->win32.frameAction = GLFW_TRUE;
557  }
558 
559  break;
560  }
561 
562  case WM_CAPTURECHANGED:
563  {
564  // HACK: Disable the cursor once the caption button action has been
565  // completed or cancelled
566  if (lParam == 0 && window->win32.frameAction)
567  {
568  if (window->cursorMode == GLFW_CURSOR_DISABLED)
569  disableCursor(window);
570 
571  window->win32.frameAction = GLFW_FALSE;
572  }
573 
574  break;
575  }
576 
577  case WM_SETFOCUS:
578  {
579  _glfwInputWindowFocus(window, GLFW_TRUE);
580 
581  // HACK: Do not disable cursor while the user is interacting with
582  // a caption button
583  if (window->win32.frameAction)
584  break;
585 
586  if (window->cursorMode == GLFW_CURSOR_DISABLED)
587  disableCursor(window);
588 
589  return 0;
590  }
591 
592  case WM_KILLFOCUS:
593  {
594  if (window->cursorMode == GLFW_CURSOR_DISABLED)
595  enableCursor(window);
596 
597  if (window->monitor && window->autoIconify)
598  _glfwIconifyWindowWin32(window);
599 
600  _glfwInputWindowFocus(window, GLFW_FALSE);
601  return 0;
602  }
603 
604  case WM_SYSCOMMAND:
605  {
606  switch (wParam & 0xfff0)
607  {
608  case SC_SCREENSAVE:
609  case SC_MONITORPOWER:
610  {
611  if (window->monitor)
612  {
613  // We are running in full screen mode, so disallow
614  // screen saver and screen blanking
615  return 0;
616  }
617  else
618  break;
619  }
620 
621  // User trying to access application menu using ALT?
622  case SC_KEYMENU:
623  {
624  if (!window->win32.keymenu)
625  return 0;
626 
627  break;
628  }
629  }
630  break;
631  }
632 
633  case WM_CLOSE:
634  {
635  _glfwInputWindowCloseRequest(window);
636  return 0;
637  }
638 
639  case WM_INPUTLANGCHANGE:
640  {
641  _glfwUpdateKeyNamesWin32();
642  break;
643  }
644 
645  case WM_CHAR:
646  case WM_SYSCHAR:
647  {
648  if (wParam >= 0xd800 && wParam <= 0xdbff)
649  window->win32.highSurrogate = (WCHAR) wParam;
650  else
651  {
652  uint32_t codepoint = 0;
653 
654  if (wParam >= 0xdc00 && wParam <= 0xdfff)
655  {
656  if (window->win32.highSurrogate)
657  {
658  codepoint += (window->win32.highSurrogate - 0xd800) << 10;
659  codepoint += (WCHAR) wParam - 0xdc00;
660  codepoint += 0x10000;
661  }
662  }
663  else
664  codepoint = (WCHAR) wParam;
665 
666  window->win32.highSurrogate = 0;
667  _glfwInputChar(window, codepoint, getKeyMods(), uMsg != WM_SYSCHAR);
668  }
669 
670  if (uMsg == WM_SYSCHAR && window->win32.keymenu)
671  break;
672 
673  return 0;
674  }
675 
676  case WM_UNICHAR:
677  {
678  if (wParam == UNICODE_NOCHAR)
679  {
680  // WM_UNICHAR is not sent by Windows, but is sent by some
681  // third-party input method engine
682  // Returning TRUE here announces support for this message
683  return TRUE;
684  }
685 
686  _glfwInputChar(window, (uint32_t) wParam, getKeyMods(), GLFW_TRUE);
687  return 0;
688  }
689 
690  case WM_KEYDOWN:
691  case WM_SYSKEYDOWN:
692  case WM_KEYUP:
693  case WM_SYSKEYUP:
694  {
695  int key, scancode;
696  const int action = (HIWORD(lParam) & KF_UP) ? GLFW_RELEASE : GLFW_PRESS;
697  const int mods = getKeyMods();
698 
699  scancode = (HIWORD(lParam) & (KF_EXTENDED | 0xff));
700  if (!scancode)
701  {
702  // NOTE: Some synthetic key messages have a scancode of zero
703  // HACK: Map the virtual key back to a usable scancode
704  scancode = MapVirtualKeyW((UINT) wParam, MAPVK_VK_TO_VSC);
705  }
706 
707  key = _glfw.win32.keycodes[scancode];
708 
709  // The Ctrl keys require special handling
710  if (wParam == VK_CONTROL)
711  {
712  if (HIWORD(lParam) & KF_EXTENDED)
713  {
714  // Right side keys have the extended key bit set
715  key = GLFW_KEY_RIGHT_CONTROL;
716  }
717  else
718  {
719  // NOTE: Alt Gr sends Left Ctrl followed by Right Alt
720  // HACK: We only want one event for Alt Gr, so if we detect
721  // this sequence we discard this Left Ctrl message now
722  // and later report Right Alt normally
723  MSG next;
724  const DWORD time = GetMessageTime();
725 
726  if (PeekMessageW(&next, NULL, 0, 0, PM_NOREMOVE))
727  {
728  if (next.message == WM_KEYDOWN ||
729  next.message == WM_SYSKEYDOWN ||
730  next.message == WM_KEYUP ||
731  next.message == WM_SYSKEYUP)
732  {
733  if (next.wParam == VK_MENU &&
734  (HIWORD(next.lParam) & KF_EXTENDED) &&
735  next.time == time)
736  {
737  // Next message is Right Alt down so discard this
738  break;

1228

1261  _GLFW_WNDCLASSNAME,
1262  wideTitle,
1263  style,
1264  xpos, ypos,

1268  GetModuleHandleW(NULL),
1269  (LPVOID) wndconfig);

1299  {

1320  AdjustWindowRectExForDpi(&rect, style, FALSE, exStyle,

1342

1957 void _glfwPollEventsWin32(void)
1958 {
1959  MSG msg;
1960  HWND handle;
1961  _GLFWwindow* window;
1962 
1963  while (PeekMessageW(&msg, NULL, 0, 0, PM_REMOVE))
1964  {
1965  if (msg.message == WM_QUIT)
1966  {
1967  // NOTE: While GLFW does not itself post WM_QUIT, other processes
1968  // may post it to this one, for example Task Manager
1969  // HACK: Treat WM_QUIT as a close on all windows
1970 
1971  window = _glfw.windowListHead;
1972  while (window)
1973  {
1974  _glfwInputWindowCloseRequest(window);
1975  window = window->next;
1976  }
1977  }
1978  else
1979  {
1980  TranslateMessage(&msg);
1981  DispatchMessageW(&msg);
1982  }
1983  }
1984 
1985  // HACK: Release modifier keys that the system did not emit KEYUP for
1986  // NOTE: Shift keys on Windows tend to "stick" when both are pressed as
1987  // no key up message is generated by the first key release
1988  // NOTE: Windows key is not reported as released by the Win+V hotkey
1989  // Other Win hotkeys are handled implicitly by _glfwInputWindowFocus
1990  // because they change the input focus
1991  // NOTE: The other half of this is in the WM_*KEY* handler in windowProc
1992  handle = GetActiveWindow();
1993  if (handle)
1994  {
1995  window = GetPropW(handle, L"GLFW");
1996  if (window)
1997  {
1998  int i;
1999  const int keys[4][2] =
2000  {
2001  { VK_LSHIFT, GLFW_KEY_LEFT_SHIFT },
2002  { VK_RSHIFT, GLFW_KEY_RIGHT_SHIFT },
2003  { VK_LWIN, GLFW_KEY_LEFT_SUPER },
2004  { VK_RWIN, GLFW_KEY_RIGHT_SUPER }
2005  };
2006 
2007  for (i = 0; i < 4; i++)
2008  {
2009  const int vk = keys[i][0];
2010  const int key = keys[i][1];
2011  const int scancode = _glfw.win32.scancodes[key];
2012 
2013  if ((GetKeyState(vk) & 0x8000))
2014  continue;
2015  if (window->keys[key] != GLFW_PRESS)
2016  continue;
2017 
2018  _glfwInputKey(window, key, scancode, GLFW_RELEASE, getKeyMods());
2019  }
2020  }
2021  }
2022 
2023  window = _glfw.win32.disabledCursorWindow;
2024  if (window)
2025  {
2026  int width, height;
2027  _glfwGetWindowSizeWin32(window, &width, &height);
2028 
2029  // NOTE: Re-center the cursor only if it has moved since the last call,
2030  // to avoid breaking glfwWaitEvents with WM_MOUSEMOVE
2031  if (window->win32.lastCursorPosX != width / 2 ||
2032  window->win32.lastCursorPosY != height / 2)
2033  {
2034  _glfwSetCursorPosWin32(window, width / 2, height / 2);
2035  }
2036  }
2037 }
