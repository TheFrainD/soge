63 /*
64  * Create a new, unique, anonymous file of the given size, and
65  * return the file descriptor for it. The file descriptor is set
66  * CLOEXEC. The file is immediately suitable for mmap()'ing
67  * the given size at offset zero.
68  *
69  * The file should not have a permanent backing store like a disk,
70  * but may have if XDG_RUNTIME_DIR is not properly implemented in OS.
71  *
72  * The file name is deleted from the file system.
73  *
74  * The file is suitable for buffer sharing between processes by
75  * transmitting the file descriptor over Unix sockets using the
76  * SCM_RIGHTS methods.
77  *
78  * posix_fallocate() is used to guarantee that disk space is available
79  * for the file at the given size. If disk space is insufficient, errno
80  * is set to ENOSPC. If posix_fallocate() is not supported, program may
81  * receive SIGBUS on accessing mmap()'ed file contents instead.
82  */
83 static int createAnonymousFile(off_t size)
84 {
85  static const char template[] = "/glfw-shared-XXXXXX";
86  const char* path;
87  char* name;
88  int fd;
89  int ret;
90 
91 #ifdef HAVE_MEMFD_CREATE
92  fd = memfd_create("glfw-shared", MFD_CLOEXEC | MFD_ALLOW_SEALING);
93  if (fd >= 0)
94  {
95  // We can add this seal before calling posix_fallocate(), as the file
96  // is currently zero-sized anyway.
97  //
98  // There is also no need to check for the return value, we couldnâ€™t do
99  // anything with it anyway.
100  fcntl(fd, F_ADD_SEALS, F_SEAL_SHRINK | F_SEAL_SEAL);
101  }
102  else
103 #elif defined(SHM_ANON)
104  fd = shm_open(SHM_ANON, O_RDWR | O_CLOEXEC, 0600);
105  if (fd < 0)
106 #endif
107  {
108  path = getenv("XDG_RUNTIME_DIR");
109  if (!path)
110  {
111  errno = ENOENT;
112  return -1;
113  }
114 
115  name = _glfw_calloc(strlen(path) + sizeof(template), 1);
116  strcpy(name, path);
117  strcat(name, template);
118 
119  fd = createTmpfileCloexec(name);
120  _glfw_free(name);
121  if (fd < 0)
122  return -1;
123  }
124 
125 #if defined(SHM_ANON)
126  // posix_fallocate does not work on SHM descriptors
127  ret = ftruncate(fd, size);
128 #else
129  ret = posix_fallocate(fd, 0, size);
130 #endif
131  if (ret != 0)
132  {
133  close(fd);
134  errno = ret;
135  return -1;
136  }
137  return fd;
138 }

140 static struct wl_buffer* createShmBuffer(const GLFWimage* image)
141 {
142  struct wl_shm_pool* pool;
143  struct wl_buffer* buffer;
144  int stride = image->width * 4;
145  int length = image->width * image->height * 4;
146  void* data;
147 
148  const int fd = createAnonymousFile(length);
149  if (fd < 0)
150  {
151  _glfwInputError(GLFW_PLATFORM_ERROR,
152  "Wayland: Creating a buffer file for %d B failed: %s",
153  length, strerror(errno));
154  return NULL;
155  }
156 
157  data = mmap(NULL, length, PROT_READ | PROT_WRITE, MAP_SHARED, fd, 0);
158  if (data == MAP_FAILED)
159  {
160  _glfwInputError(GLFW_PLATFORM_ERROR,
161  "Wayland: mmap failed: %s", strerror(errno));
162  close(fd);
163  return NULL;
164  }
165 
166  pool = wl_shm_create_pool(_glfw.wl.shm, fd, length);
167 
168  close(fd);
169  unsigned char* source = (unsigned char*) image->pixels;
170  unsigned char* target = data;
171  for (int i = 0; i < image->width * image->height; i++, source += 4)
172  {
173  unsigned int alpha = source[3];
174 
175  *target++ = (unsigned char) ((source[2] * alpha) / 255);
176  *target++ = (unsigned char) ((source[1] * alpha) / 255);
177  *target++ = (unsigned char) ((source[0] * alpha) / 255);
178  *target++ = (unsigned char) alpha;
179  }
180 
181  buffer =
182  wl_shm_pool_create_buffer(pool, 0,
183  image->width,
184  image->height,
185  stride, WL_SHM_FORMAT_ARGB8888);
186  munmap(data, length);
187  wl_shm_pool_destroy(pool);
188 
189  return buffer;
190 }

346 static void checkScaleChange(_GLFWwindow* window)
347 {
348  int scale = 1;
349  int monitorScale;
350 
351  // Check if we will be able to set the buffer scale or not.
352  if (_glfw.wl.compositorVersion < 3)
353  return;
354 
355  // Get the scale factor from the highest scale monitor.
356  for (int i = 0; i < window->wl.monitorsCount; ++i)
357  {
358  monitorScale = window->wl.monitors[i]->wl.scale;
359  if (scale < monitorScale)
360  scale = monitorScale;
361  }
362 
363  // Only change the framebuffer size if the scale changed.
364  if (scale != window->wl.scale)
365  {
366  window->wl.scale = scale;
367  wl_surface_set_buffer_scale(window->wl.surface, scale);
368  resizeWindow(window);
369  }

720 static void handleEvents(int timeout)
721 {
722  struct pollfd fds[] =
723  {
724  { wl_display_get_fd(_glfw.wl.display), POLLIN },
725  { _glfw.wl.timerfd, POLLIN },
726  { _glfw.wl.cursorTimerfd, POLLIN },
727  };
728 
729  while (wl_display_prepare_read(_glfw.wl.display) != 0)
730  wl_display_dispatch_pending(_glfw.wl.display);
731 
732  // If an error other than EAGAIN happens, we have likely been disconnected
733  // from the Wayland session; try to handle that the best we can.
734  if (wl_display_flush(_glfw.wl.display) < 0 && errno != EAGAIN)
735  {
736  _GLFWwindow* window = _glfw.windowListHead;
737  while (window)
738  {
739  _glfwInputWindowCloseRequest(window);
740  window = window->next;
741  }
742 
743  wl_display_cancel_read(_glfw.wl.display);
744  return;
745  }
746 
747  if (poll(fds, 3, timeout) > 0)
748  {
749  if (fds[0].revents & POLLIN)
750  {
751  wl_display_read_events(_glfw.wl.display);
752  wl_display_dispatch_pending(_glfw.wl.display);
753  }
754  else
755  wl_display_cancel_read(_glfw.wl.display);
756 
757  if (fds[1].revents & POLLIN)
758  {
759  uint64_t repeats;
760 
761  if (read(_glfw.wl.timerfd, &repeats, sizeof(repeats)) == 8)
762  {
763  for (uint64_t i = 0; i < repeats; i++)
764  {
765  _glfwInputKey(_glfw.wl.keyboardFocus,
766  _glfw.wl.keyboardLastKey,
767  _glfw.wl.keyboardLastScancode,
768  GLFW_PRESS,
769  _glfw.wl.xkb.modifiers);
770  _glfwInputTextWayland(_glfw.wl.keyboardFocus,
771  _glfw.wl.keyboardLastScancode);
772  }
773  }
774  }
775 
776  if (fds[2].revents & POLLIN)
777  {
778  uint64_t repeats;
779 
780  if (read(_glfw.wl.cursorTimerfd, &repeats, sizeof(repeats)) == 8)
781  incrementCursorImage(_glfw.wl.pointerFocus);
782  }
783  }
784  else
785  wl_display_cancel_read(_glfw.wl.display);
786 }
